六大函数:

\begin{itemize}
\item 
默认构造函数: 
\begin{lstlisting}[style=styleCXX]
X()
\end{lstlisting}

\item 
复制构造函数:
\begin{lstlisting}[style=styleCXX]
 X(const X&)
\end{lstlisting}

\item 
复制赋值操作符: 
\begin{lstlisting}[style=styleCXX]
X& operator = (const X&)
\end{lstlisting}

\item 
移动构造函数: 
\begin{lstlisting}[style=styleCXX]
X(X&&)
\end{lstlisting}

\item 
移动赋值操作符: 
\begin{lstlisting}[style=styleCXX]
X& operator = (X&&)
\end{lstlisting}

\item 
析构函数: 
\begin{lstlisting}[style=styleCXX]
~X()
\end{lstlisting}
\end{itemize}
