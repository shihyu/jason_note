report.display_mae_mfe_analysis()

еҰӮдҪ•и§Ји®Җең–зө„

е°ҮдәӨжҳ“еҲҶзӮәзҚІеҲ© (profit-и—Қй»һ) иҲҮиҷ§жҗҚ (loss-зҙ…й»һ) еҲҶеҲҘе‘ҲзҸҫпјҢең–зө„еҸіж–№зҡ„legendеҸҜд»Ҙд»»дёҖй»һйҒёпјҢеҸӘзңӢprofitжҲ–lossзҡ„еҲҶзҫӨе‘ҲзҸҫгҖӮжҺҘи‘—жңғгҖҢз”ұдёҠеҲ°дёӢгҖҒз”ұе·ҰиҮіеҸігҖҚпјҢи§ЈйҮӢеҗ„еӯҗең–з”ЁйҖ”гҖӮ

е ұй…¬зҺҮзөұиЁҲең–

еӯҗең–1-1пјҢе‘ҲзҸҫзӯ–з•ҘдёӢжҜҸзӯҶдәӨжҳ“зҡ„е ұй…¬зҺҮеҲҶдҪҲпјҢиЁҲз®—еҮәеӢқзҺҮеҸҠе№іеқҮжҜҸзӯҶе ұй…¬гҖӮ ең–зүҮжЁҷйЎҢзӮәдәӨжҳ“еӢқзҺҮпјҢз¶ иүІиҷӣз·ҡзӮәе№іеқҮжҜҸзӯҶдәӨжҳ“зҡ„е ұй…¬зҺҮгҖӮ еҲҶдҪҲжӣІз·ҡи¶Ҡе№іеқҰпјҢд»ЈиЎЁе ұй…¬зҺҮзҜ„еңҚеӨ§пјҢеҸҜиғҪжңүијғеӨҡзҡ„жҘөз«Ҝе ұй…¬зҺҮиҰҒиҷ•зҗҶпјҢйҖҡеёёеҮәзҸҫеңЁжіўеӢ•еӨ§зҡ„зӯ–з•ҘгҖӮ иӢҘе‘ҲзҸҫеҸіеҒҸеһӢж…Ӣ(еҸіеҒҙзҡ„е°ҫйғЁжӣҙй•·пјҢеҲҶдҪҲзҡ„дё»й«”йӣҶдёӯеңЁе·ҰеҒҙ)пјҢд»ЈиЎЁеӨҡж•ёдәӨжҳ“зӮәиҷ§жҗҚпјҢиӢҘж•ҙй«”зӯ–з•ҘзӮәзҚІеҲ©пјҢеүҮзҚІеҲ©зӮәе°‘зӯҶдәӨжҳ“зӮәдё»иҰҒиІўзҚ»гҖӮ иӢҘе‘ҲзҸҫе·ҰеҒҸеһӢж…Ӣ(е·ҰеҒҙзҡ„е°ҫйғЁжӣҙй•·пјҢеҲҶдҪҲзҡ„дё»й«”йӣҶдёӯеңЁеҸіеҒҙ)пјҢд»ЈиЎЁеӨҡж•ёдәӨжҳ“зӮәзҚІеҲ©гҖӮ иӢҘе‘ҲзҸҫйҗҳеһӢжӣІз·ҡпјҢд»ЈиЎЁеҲҶдҪҲијғзӮәе№іеқҮгҖӮ

Edge Ratio жҷӮеәҸең–

еӯҗең–1-2жҳҜзӯ–з•Ҙedge ratioйҡЁйҖІе ҙжҷӮй–“ (xи»ё) еҫҢзҡ„и®ҠеҢ–пјҢеҸҜд»ҘеҲӨж–·йҡЁи‘—жҢҒжңүжҷӮй–“жҺЁз§»пјҢзӯ–з•ҘжңүжІ’жңүжіўе№…ж“ҚдҪңе„ӘеӢўгҖӮ

еҸғж•ёиЁӯе®ҡ

edge ratioзҡ„иЁҲз®—иЁӯе®ҡз”ұеӣһжё¬еҮҪж•ё backtest.sim() иЈЎзҡ„mae_mfe_window, mae_mfe_window_step е…©еҖӢеҸғж•ёдҫҶжҺ§еҲ¶гҖӮ

  • mae_mfe_window : иЁҲз®—mae_mfeж–јйҖІе ҙеҫҢж–јдёҚеҗҢжҢҒжңүеӨ©ж•ёдёӢзҡ„ж•ёж“ҡи®ҠеҢ–пјҢдё»иҰҒжҮүз”ЁзӮәе„ӘеӢўжҜ”зҺҮиЁҲз®—гҖӮй җиЁӯзӮә0пјҢеҸӘжңғз”ўз”ҹеҮәе ҙйҡҺж®өзҡ„mae_mfeгҖӮ
  • mae_mfe_window_step : иҲҮ mae_mfe_windowеҸғж•ёеҒҡжҗӯй…ҚпјҢзӮәжҷӮй–“й–“йҡ”иЁӯе®ҡпјҢй җиЁӯзӮә1гҖӮиӢҘmae_mfe_windowиЁӯ20пјҢmae_mfe_window_step иЁӯе®ҡзӮә2пјҢзӣёз•¶ж–ј python зҡ„range(0,20,2)пјҢд»Ҙ2ж—ҘзӮәй–“и·қиЁҲз®—mae_mfeгҖӮ

жҮүз”Ёи§ЈйҮӢ

edge ratioиӢҘдёҖзӣҙдҝқжҢҒеңЁ1д»ҘдёҠпјҢжҢҒжңүйғҪе…·жңүе„ӘеӢўпјҢеӯҗең–зҜ„дҫӢе°ұжҳҜйҖҷйЎһжғ…жіҒпјҢй–ӢеұҖе°ұжңүдёҚйҢҜиЎЁзҸҫпјҢжҳҺйЎҜзҡ„и°·еә•иҗҪеңЁз¬¬8еӨ©еҫҢжҢҒзәҢиө°й«ҳпјҢд»ЈиЎЁеҸҜиғҪе»¶йҒІеҲ°з¬¬8еӨ©йҖІе ҙжңғжңүдҪҺй»һпјҢд№ӢеҫҢзҚІеҲ©дёҖи·Ҝж”ҫеӨ§е„ӘеӢўгҖӮ edge ratioжҷӮеәҸең–иө°еӢўжңүеҫҲеӨҡзЁ®пјҢиӢҘжҳҜй–ӢдҪҺиө°й«ҳпјҢдёҖй–Ӣе§ӢйғҪдҪҺж–ј1пјҢд»ЈиЎЁзӯ–з•ҘеҸҜиғҪеӨӘж—©йҖІе ҙпјҢдёҖй–Ӣе§ӢйғҪиҰҒе…ҲжүҝеҸ—иҷ§жҗҚпјҢйҖҷжҷӮеҸҜд»ҘжӘўиЁҺйҖІе ҙжҷӮж©ҹй»һпјҢиҖғж…®е»¶йҒІйҖІе ҙгҖӮ

иӢҘedge ratioиө°еӢўдҝқжҢҒеңЁ1д»ҘдёҠпјҢд»ЈиЎЁзӯ–з•Ҙе„ӘеӢўжҳҺйЎҜгҖӮиӢҘйӮ„йҡЁи‘—жҷӮй–“иө°й«ҳпјҢзҚІеҲ©з©әй–“д№ҹдёҠеҚҮпјҢзӯ–з•Ҙе®№йҢҜзҺҮе°ұијғеӨ§пјҢе°ұз®—еӣ дёҖдәӣеӣ зҙ е»¶йҒІйҖІе ҙд»ҚжңүијғеӨ§ж©ҹзҺҮжңүзҚІеҲ©зҜ„еңҚгҖӮ иӢҘedge ratioиө°еӢўеҫҲеёёеңЁ1д»ҘдёӢпјҢд»ЈиЎЁзӯ–з•Ҙз¶“еёёиў«иҷ§жҗҚеЈ“и‘—жү“пјҢжҳҜзӯ–з•ҘиІ йқўиЁҠиҷҹгҖӮ

иӢҘйҡЁжҢҒжңүжҷӮй–“и®ҠеҢ–пјҢе„ӘеӢўжјёжјёжөҒеӨұпјҢжҜ”зҺҮй–Ӣе§ӢдёӢйҷҚпјҢд»ЈиЎЁMAEжҷ®йҒҚи®Ҡй«ҳпјҢеҸҜиғҪжҳҜзӯ–з•ҘеӮ¬еҢ–еҠ‘иӨӘиүІпјҢи©Ізӯ–з•ҘйҒ©еҗҲзҹӯз·ҡж“ҚдҪңдёҰиҖғж…®еҠ дёҠеҒңеҲ©жҸҗж—©еҮәе ҙгҖӮ иӢҘedge ratioиө°еӢўи·іеӢ•пјҢд»ЈиЎЁз„ЎжҳҺйЎҜи¶ЁеӢўеҸҜеҲӨж–·гҖӮ иӢҘзӯ–з•ҘйҖұжңҹжҳҜ20еӨ©пјҢзҷјзҸҫtime_scaleеӨ§ж–ј20жҷӮпјҢedge ratioи¶ЁеӢўжҢҒзәҢиө°еҚҮпјҢеүҮйҖҸйңІзӯ–з•ҘеҸҜиғҪеӨӘж—©еҮәе ҙпјҢйҢҜйҒҺеҫҢйқўжӣҙеӨ§зҡ„е ұй…¬пјҢеҸҜд»ҘиҖғж…®дҝ®жӯЈжҢҒиӮЎйҖұжңҹпјҢеҗғеҲ°жӣҙеӨ§зҡ„зҚІеҲ©гҖӮ

MAE/Return еҲҶдҪҲең–

еӯҗең–1-3пјҢxи»ёзӮәе ұй…¬зҺҮпјҢYи»ёзӮәMAEпјҢе°ҮеӢқж•—жүӢеҲҶзҫӨйЎҜзӨәжҲҗж•Јй»һең–пјҢжҜ”е°Қе ұй…¬зҺҮиҲҮMAEзҡ„й—ңдҝӮгҖӮ жӯӨзҜ„дҫӢдёӯеҸҜд»ҘзҷјзҸҫеӨҡж•ёзҚІеҲ©зҡ„и—Қй»һйғҪжңүијғе°Ҹзҡ„MAEпјҢиҷ§жҗҚзҡ„зҙ…й»һжңүијғеӨ§зҡ„MAEгҖӮ иҷ§жҗҚйғЁдҪҚзҡ„MAEз¬¬75%дҪҚж•ёзӮә10.77%пјҢе№ҫд№ҺжүҖжңүзҡ„и—Қй»һйғҪдҪҺж–јйҖҷеҖӢдҪҚзҪ®пјҢд№ҹе°ұжҳҜиӘӘйҒҺдәҶйҖҷеҖӢдҪҚзҪ®пјҢдәӨжҳ“жңҖзөӮе°ұе®№жҳ“жҳҜиҷ§жҗҚзөҗжһңпјҢеҸҜиЁӯзӮәеҒңжҗҚеҸғиҖғдҪҚзҪ®пјҢеҸҜдҝқз•ҷеӨҡж•ёзҚІеҲ©йғЁдҪҚгҖҒжёӣе°‘еӨ§иі йғЁдҪҚжҗҚеӨұгҖӮ зҚІеҲ©йғЁдҪҚзҡ„MAEз¬¬75%дҪҚж•ёзӮә2.93%пјҢд»ЈиЎЁеӨҡж•ёзҚІеҲ©йғЁдҪҚеңЁжҢҒжңүйҒҺзЁӢдёӯеҸҜиғҪзҡ„жңҖдҪҺй»һеҚҖй–“пјҢзў°еҲ°йҖҷдҪҚзҪ®еҫҢе°ұжңүијғй«ҳж©ҹзҺҮеҶҚеҫҖдёҠпјҢз©ҚжҘөж“ҚдҪңиҖ…жҲ–еҲҶжү№йҖІе ҙиҖ…еҸҜиЁӯзӮәж”Өе№іеҠ зўјй»һдҪҚзҪ®пјҢжңүж©ҹжңғи®“зҚІеҲ©з©әй–“жӣҙеӨҡжҲ–иі жӣҙеӨҡгҖӮ

MFE/MAE еҲҶдҪҲең–

еҫһеӯҗең– 2-1гҖҒ2-2 еҸҜд»Ҙи§ҖеҜҹMAEиҲҮMFEзҡ„ж•ёж“ҡй—ңдҝӮпјҢж•Јй»һең–еӨ§е°Ҹз”ұе ұй…¬зҺҮдҫҶжұәе®ҡгҖӮ жҜ”е°Қе…©ејөең–еҸҜзҷјзҸҫпјҢзӯ–з•Ҙе…§иЁұеӨҡGMFEеҫҲеӨ§зҡ„жЁҷзҡ„пјҢйғҪжҜ”BMFEеӨ§пјҢд»ЈиЎЁиЁұеӨҡжјІе№…йғҪзҷјз”ҹеңЁMAEд№ӢеҫҢгҖӮжғіиҰҒжңүијғй«ҳзҚІеҲ©пјҢе°ұиҰҒе…ҲеҝҚеҸ—еӣһжӘ”пјҢйҖҡеёёйҖҷе®№жҳ“зҷјз”ҹеңЁи¶ЁеӢўжіўж®өзӯ–з•ҘгҖӮ иӢҘжҳҜзҹӯз·ҡе„Әз•°зҡ„зӯ–з•ҘпјҢBMFE жңғжҜ”ијғй«ҳпјҢеҸҜд»Ҙжңүијғй«ҳж©ҹзҺҮеңЁжҺҘи§ёMAEжҲ–еҒңжҗҚеүҚе…ҲеҒҡеҒңеҲ©гҖӮ

еҲҶдҪҲиұЎйҷҗең–и§Ј

MAE/BMFEеҲҶдҪҲең–(иЁ»3) иғҪе№«еҠ©жҲ‘еҖ‘зңӢеҮәзӯ–з•Ҙй«”иіӘгҖҒе„ӘеҢ–иЁӯзҪ®еҒңжҗҚеҒңеҲ©гҖӮ еӨ§еҺҹеүҮжҳҜгҖҢеҲҶдҪҲеңЁз¬¬дәҢиұЎйҷҗзҡ„й»һи¶ҠеӨҡи¶ҠеҘҪпјҢ з¬¬еӣӣиұЎйҷҗзҡ„й»һи¶Ҡе°‘и¶ҠеҘҪгҖҚгҖҒгҖҢзҚІеҲ©иҲҮиҷ§жҗҚжҳҺйЎҜеҲҶзҫӨеңЁдёҚеҗҢиұЎйҷҗгҖҚгҖӮ еҰӮжӯӨ stop_lossйҒҺжҝҫжҺүеӨҡж•ёmaeйҒҺеӨ§зҡ„жЁҷзҡ„пјҢе°‘йҒҺжҝҫжҺүзҚІеҲ©зҡ„жЁҷзҡ„гҖӮtake_profitзӣЎйҮҸи®“еӨҡж•ёиҷ§жҗҚзҡ„дәӨжҳ“е…Ҳи§ёеҸҠеҒңеҲ©еҮәе ҙгҖӮ

MDD/GMFE еҲҶдҪҲең–

еӯҗең–2-3пјҢеҲӨж–·жҗҚзӣҠе…©е№ій»һиҲҮйҺ–еҲ©й»һпјҢж©ҳз·ҡзӮә45еәҰз·ҡгҖӮ ж©ҳз·ҡд»ҘдёҠзӮәMDD > GMFEпјҢеҰӮжһңи¶ҠеӨҡзҚІеҲ©й»һдҪҚж–јйҖҷеҖӢдҪҚзҪ®пјҢд»ЈиЎЁжҢҒжңүжӯ·зЁӢеҸҜиғҪжӯ·з¶“еӨ§еӣһжӘ”еҗғжҺүзҚІеҲ©иҪүзӮәиҷ§жҗҚпјҢйӣ–з„¶жңҖзөӮжңғжҳҜзҚІеҲ©пјҢдҪҶжҲ‘еҖ‘еҺҹжң¬жңүж©ҹжңғиіәжӣҙеӨҡгҖӮ

MDD > GMFE зҡ„жғ…жіҒеёёжҳҜдёҖй–Ӣе§Ӣе°ұеҗғеӨ§иҷ§жҗҚпҪһеҫҢдҫҶиҪүжӯЈпјҢжҲ–жҳҜйҖ”дёӯеӨ§иіәеҫҢпјҢзӘҒз„¶жҖҘйҖҹдёӢж®әиі йҢўгҖӮйғҪжҳҜжҜ”ијғдёҚзҗҶжғізҡ„зӢҖжіҒгҖӮеӯҗжЁҷйЎҢйЎҜзӨәзҡ„гҖҢMissed win-profits pctгҖҚзӮәгҖҢзҚІеҲ©дәӨжҳ“дҪҚж–јж©ҳз·ҡдёҠзҡ„ж•ёйҮҸ/зҚІеҲ©дәӨжҳ“ж•ёгҖҚпјҢж•ёеҖји¶Ҡй«ҳд»ЈиЎЁжҪӣеңЁйҢҜеӨұзҚІеҲ©зҡ„ж©ҹжңғијғй«ҳпјҢж•ёеҖји¶Ҡй«ҳд»ЈиЎЁи¶ҠйңҖиҰҒиЁӯе®ҡз§»еӢ•еҒңеҲ©еҺ»дҝқиӯ·зҚІеҲ©гҖӮ

def get_mae(
    order_type: str,
    entry_time: dt.datetime or int,
    exit_time: dt.datetime or int,
    price_data: pd.DataFrame or pd.Series or np.array or list,
):
    """
    parameters:
        order_type: str, еҒҡеӨҡжҲ–еҒҡз©ә ('long' жҲ– 'short')
        entry_time: dt.datetime жҲ– int, йҖІе ҙжҷӮй–“ (иӢҘжңүOHLCиіҮж–ҷпјҢеүҮиҰ–зӮәopenйҖІе ҙ)
        exit_time: dt.datetime жҲ– int, еҮәе ҙжҷӮй–“ (иӢҘжңүOHLCиіҮж–ҷпјҢеүҮиҰ–зӮәcloseеҮәе ҙ)
        price_data: pd.DataFrame жҲ– pd.Series, д»ҘdatetimeзӮәindexзҡ„еғ№ж јиіҮж–ҷ
                    np.array жҲ– list, д»ҘйҷЈеҲ—зҙҖйҢ„зҡ„еғ№ж јиіҮж–ҷ
                    (DataFrameзӮәOHLCиіҮж–ҷпјӣSeriesгҖҒArrayжҲ–ListеүҮзӮәTickжҲ–CloseиіҮж–ҷ)

    return:
        mae: float, дәӨжҳ“жңҹй–“жңҖеӨ§еӣһеҫ№
        mae_time: dt.datetime жҲ– int, MAEзҷјз”ҹжҷӮй–“
    """
    mae = 0.0
    mae_time = None

    if type(price_data) == pd.DataFrame:
        data = price_data.loc[entry_time:exit_time]
        entry_price = data["open"].loc[entry_time]
        for index, values in data.iterrows():
            if order_type == "long":
                drawdown = values["low"] - entry_price
            elif order_type == "short":
                drawdown = entry_price - values["high"]
            if mae > drawdown:
                mae = drawdown
                mae_time = index

    elif type(price_data) == pd.Series:
        data = price_data.loc[entry_time:exit_time]
        entry_price = data.loc[entry_time]
        for index, values in enumerate(data):
            if order_type == "long":
                drawdown = values - entry_price
            elif order_type == "short":
                drawdown = entry_price - values
            if mae > drawdown:
                mae = drawdown
                mae_time = index

    elif type(price_data) == np.array or type(price_data) == list:
        entry_price = price_data[entry_time]
        for index, values in enumerate(price_data[entry_time : exit_time + 1]):
            if order_type == "long":
                drawdown = values - entry_price
            elif order_type == "short":
                drawdown = entry_price - values
            if mae > drawdown:
                mae = drawdown
                mae_time = entry_time + index

    return mae, mae_time

def mae_mfe_pair(
    order: list,
    price_data: pd.DataFrame or pd.Series or np.array or list,
    mae_first: bool = True,
):
    """
    parameters:
        order: list, жүҖжңүдәӨжҳ“зҙҖйҢ„пјҢдәӨжҳ“зҙҖйҢ„зӮәdict
                дәӨжҳ“зҙҖйҢ„ж јејҸ:
                {
                    'order_type': str,
                    'entry_time': dt.datetime жҲ– int,
                    'exit_time': dt.datetime жҲ– int
                }
        price_data: pd.DataFrame жҲ– pd.Series, д»ҘdatetimeзӮәindexзҡ„еғ№ж јиіҮж–ҷ
                    np.array жҲ– list, д»ҘйҷЈеҲ—зҙҖйҢ„зҡ„еғ№ж јиіҮж–ҷ
                    (DataFrameзӮәOHLCиіҮж–ҷпјӣSeriesгҖҒArrayжҲ–ListеүҮзӮәTickжҲ–CloseиіҮж–ҷ)
        mae_first: bool (й җиЁӯзӮәTrue), MFEжҳҜеҗҰеҮәзҸҫеңЁMAEд№ӢеүҚ

    return:
        results: list, жүҖжңүдәӨжҳ“зҙҖйҢ„зөҗжһңпјҢдәӨжҳ“зҙҖйҢ„зөҗжһңзӮәdict
                дәӨжҳ“зҙҖйҢ„зөҗжһңж јејҸ:
                {
                    'mae': float,
                    'mfe': float,
                    'mae_time': dt.datetime жҲ– int,
                    'mfe_time': dt.datetime жҲ– int
                }
    """
    results = []
    for this_order in order:
        mae, mae_time = get_mae(
            this_order["order_type"],
            this_order["entry_time"],
            this_order["exit_time"],
            price_data,
        )

        if mae_first:
            mfe, mfe_time = get_mfe(
                this_order["order_type"], this_order["entry_time"], mae_time, price_data
            )
        else:
            mfe, mfe_time = get_mfe(
                this_order["order_type"],
                this_order["entry_time"],
                this_order["exit_time"],
                price_data,
            )

        results.append(
            {"mae": mae, "mfe": mfe, "mae_time": mae_time, "mfe_time": mfe_time}
        )

    return results


def mae_mfe(orders, x_y_line=True, return_fig_ax=False):
    """
    parameters:
        orders: list, жүҖжңүдәӨжҳ“зҡ„maeе’ҢmfeиіҮж–ҷпјҢиіҮж–ҷеҝ…й Ҳж”ҫеңЁdictиЈЎйқў
            иіҮж–ҷж јејҸ:
            {
                'mae': float,
                'mfe': float,
                ...
                (еҸҜд»ҘеҢ…еҗ«е…¶д»–иіҮж–ҷ)
            }
        
        x_y_line: bool (й җиЁӯзӮәTrue), жҳҜеҗҰз№ӘиЈҪx=yзҡ„иҷӣз·ҡ
        return_fig_ax: bool (й җиЁӯзӮәFalse), жҳҜеҗҰеӣһеӮіmatplotlibзҡ„з№Әең–е…ғд»¶

    return:
        (optional)
        fig: plotlyзҡ„еҹәжң¬з№Әең–е…ғд»¶
    """
    fig = go.Figure()

    mae = []
    mfe = []
    for order in orders:
        mae.append(abs(order["mae"]))
        mfe.append(order["mfe"])
    fig.add_trace(
        go.Scatter(
            x=mae, y=mfe, mode="markers", name="Orders", marker=dict(color="red")
        )
    )

    if x_y_line:
        x_y_line_x = [0, max(max(mae), max(mfe))]
        x_y_line_y = [0, max(max(mae), max(mfe))]
        fig.add_trace(
            go.Scatter(
                x=x_y_line_x,
                y=x_y_line_y,
                mode="lines",
                name="x=y",
                line=dict(dash="dash", color="black", width=1),
            )
        )

    fig.update_layout(
        xaxis_title="MAE", yaxis_title="MFE", title="MAE vs MFE", legend=dict(title="")
    )

    if return_fig_ax:
        return fig
    else:
        pio.show(fig)

import pandas as pd


def get_metrics(prices):
    base_price = prices[0]
    min_price = min(prices)
    max_price = max(prices[1:])

    max_price_before_min = (
        max(prices[: prices.index(min_price)])
        if min_price is not None and prices[: prices.index(min_price)] != []
        else None
    )

    bmfe = (
        ((max_price_before_min - base_price) / base_price) * 100
        if max_price_before_min is not None and base_price < max_price_before_min
        else 0
    )

    gmfe = (
        ((max_price - base_price) / base_price) * 100
        if max_price is not None and max_price > base_price
        else 0
    )

    mae = (
        (min_price - base_price) / base_price * 100
        if min_price is not None and min_price < base_price
        else 0
    )

    return mae, bmfe, gmfe

    df["Drawdown"] = df["Close"] / df["Max"] - 1

import pandas as pd

class PriceMetrics:
    def __init__(self, prices):
        self.prices = prices
        self.base_price = prices[0]
        self.min_price = min(prices)
        self.max_price = max(prices[1:])
        self.max_price_before_min = self._get_max_price_before_min()
        self.mae = self._calculate_mae()
        self.bmfe = self._calculate_bmfe()
        self.gmfe = self._calculate_gmfe()
        self.edge_ratio = self._calculate_edge_ratio()

    def _get_max_price_before_min(self):
        if self.min_price is None or self.prices[:self.prices.index(self.min_price)] == []:
            return None
        return max(self.prices[:self.prices.index(self.min_price)])

    def _calculate_mae(self):
        if self.min_price is None or self.min_price >= self.base_price:
            return 0
        return (self.min_price - self.base_price) / self.base_price * 100

    def _calculate_bmfe(self):
        if self.max_price_before_min is None or self.base_price >= self.max_price_before_min:
            return 0
        return ((self.max_price_before_min - self.base_price) / self.base_price) * 100

    def _calculate_gmfe(self):
        if self.max_price is None or self.max_price <= self.base_price:
            return 0
        return ((self.max_price - self.base_price) / self.base_price) * 100

    def _calculate_edge_ratio(self):
        if self.mae == 0:
            return 0
        return ((self.gmfe - self.mae) / abs(self.mae)) * 100

    def get_max_drawdown(self):
        df = pd.DataFrame(self.prices, columns=["Close"])
        df["Max"] = df["Close"].cummax()
        df["Drawdown"] = df["Close"] / df["Max"] - 1
        df["Max Drawdown"] = df["Drawdown"].cummin()
        mdd = df["Max Drawdown"].min()
        return mdd * 100
