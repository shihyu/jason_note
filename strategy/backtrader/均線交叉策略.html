# sma cross strategy
class SmaCross(bt.Strategy):
    # дәӨжҳ“зҙҖйҢ„
    def log(self, txt, dt=None):
        dt = dt or self.datas[0].datetime.date(0)
        print('%s, %s' % (dt.isoformat(), txt))
    
    # иЁӯе®ҡдәӨжҳ“еҸғж•ё
    params = dict(
        ma_period_short=5,
        ma_period_long=10
    )

    def __init__(self):
        # еқҮз·ҡдәӨеҸүзӯ–з•Ҙ
        sma1 = bt.ind.SMA(period=self.p.ma_period_short)
        sma2 = bt.ind.SMA(period=self.p.ma_period_long)
        self.crossover = bt.ind.CrossOver(sma1, sma2)
        
        # дҪҝз”ЁиҮӘиЁӮзҡ„sizerеҮҪж•ёпјҢе°ҮеёідёҠзҡ„йҢўall-in
        self.setsizer(sizer())
        
        # з”Ёй–ӢзӣӨеғ№еҒҡдәӨжҳ“
        self.dataopen = self.datas[0].open

    def next(self):
        # еёіжҲ¶жІ’жңүйғЁдҪҚ
        if not self.position:
            # 5maеҫҖдёҠз©ҝи¶Ҡ20ma
            if self.crossover > 0:
                # еҚ°еҮәиІ·иіЈж—ҘжңҹиҲҮеғ№дҪҚ
                self.log('BUY ' + ', Price: ' + str(self.dataopen[0]))
                # дҪҝз”Ёй–ӢзӣӨеғ№иІ·е…ҘжЁҷзҡ„
                self.buy(price=self.dataopen[0])
        # 5maеҫҖдёӢз©ҝи¶Ҡ20ma
        elif self.crossover < 0:
            # еҚ°еҮәиІ·иіЈж—ҘжңҹиҲҮеғ№дҪҚ
            self.log('SELL ' + ', Price: ' + str(self.dataopen[0]))
            # дҪҝз”Ёй–ӢзӣӨеғ№иіЈеҮәжЁҷзҡ„
            self.close(price=self.dataopen[0])

# иЁҲз®—дәӨжҳ“йғЁдҪҚ
class sizer(bt.Sizer):
    def _getsizing(self, comminfo, cash, data, isbuy):
        if isbuy:
            return math.floor(cash/data[1])
        else:
            return self.broker.getposition(data)

# иЁҲз®—дәӨжҳ“йғЁдҪҚ
class sizer(bt.Sizer):
    def _getsizing(self, comminfo, cash, data, isbuy):
        if isbuy:
            return math.floor(cash / data[1])
        else:
            return self.broker.getposition(data)
