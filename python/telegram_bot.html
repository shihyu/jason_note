def get_daily_horoscope(sign: str, day: str) -> dict:
    """é€šéç‰¹å®šçš„æ˜Ÿåº§ç²å–é‹å‹¢ã€‚

    é—œéµå­—è§£é‡‹:
    sign:str - æ˜Ÿåº§

    response = requests.get(url, params)

    return response.json()
