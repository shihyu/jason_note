жЁЎзө„ (Module) е’ҢеҘ—д»¶ (Package)

жҲ‘еҖ‘еӯёжңғеҮҪејҸеҫҢпјҢзЁӢејҸзўјеҸҜд»ҘеҲҶйӣўпјҢз„¶иҖҢпјҢйҡЁи‘—е°ҲжЎҲиҰҸжЁЎдёҠеҚҮпјҢеҮҪејҸеҗҚзЁұжңүеҸҜиғҪзӣёдә’иЎқзӘҒгҖӮйӣ–з„¶пјҢжҲ‘еҖ‘д№ҹеҸҜд»Ҙдҝ®ж”№еҮҪејҸеҗҚзЁұпјҢдҪҶжҳҜпјҢеҸӘйқ еҮҪж•ёеҗҚзЁұдҫҶеҚҖеҲҶеҮҪејҸпјҢеҫҖеҫҖжңғйҖ жҲҗеҮҪж•ёеҗҚзЁұи®Ҡеҫ—еҶ—й•·гҖӮеғҸ C иӘһиЁҖдёӯпјҢжІ’жңүйЎҚеӨ–зҡ„ж©ҹеҲ¶иҷ•зҗҶеҮҪејҸеҗҚзЁұзҡ„иЎқзӘҒпјҢе°ұжңғзңӢеҲ°еҫҲеӨҡй•·еҗҚзЁұзҡ„еҮҪејҸпјҢеғҸжҳҜ gtk_application_get_windows (еҮәиҮӘ GTK+ еҮҪејҸеә«)гҖӮRust жҸҗдҫӣжЁЎзө„ (module) зҡ„ж©ҹеҲ¶пјҢиҷ•зҗҶеҮҪејҸе‘ҪеҗҚиЎқзӘҒзҡ„е•ҸйЎҢгҖӮ

иЁ»пјҡеңЁиЁұеӨҡзЁӢејҸиӘһиЁҖдёӯпјҢд»Ҙе‘ҪеҗҚз©әй–“ (namespace) жҸҗдҫӣйЎһдјјзҡ„ж©ҹеҲ¶гҖӮ

еңЁжҲ‘еҖ‘е…ҲеүҚзҡ„е…§е®№дёӯпјҢеҮҪејҸе’Ңдё»зЁӢејҸйғҪеҜ«еңЁеҗҢдёҖеҖӢжӘ”жЎҲгҖӮеңЁеҜҰеӢҷдёҠпјҢжҲ‘еҖ‘жңғе°ҮеҮҪејҸжҲ–зү©д»¶зҚЁз«ӢеҮәдҫҶпјҢиЈҪжҲҗеҘ—д»¶ (package)пјҢд№ӢеҫҢеҸҜд»ҘйҮҚиҰҶеҲ©з”ЁгҖӮдҫӢеҰӮпјҢеҗҢдёҖеҘ—еҮҪејҸеә«пјҢеҸҜдҫӣзөӮз«Ҝж©ҹжҲ–ең–еҪўд»ӢйқўзӯүдёҚеҗҢдҪҝз”ЁиҖ…д»ӢйқўдҫҶдҪҝз”ЁгҖӮеңЁ Rust дёӯпјҢеҘ—д»¶е’ҢжЁЎзө„зӣёдә’й—ңйҖЈгҖӮRust зҡ„еҘ—д»¶еҸҲзЁұзӮә crateгҖӮ

дҪҝз”ЁжЁЎзө„

йӣ–з„¶жҲ‘еҖ‘еңЁе…ҲеүҚзҡ„е…§е®№жІ’жңүеј·иӘҝжЁЎзө„пјҢеҜҰйҡӣдёҠпјҢжҲ‘еҖ‘е·Із¶“еңЁдҪҝз”ЁжЁЎзө„дәҶгҖӮжҲ‘еҖ‘еӣһй ӯзңӢе…ҲеүҚзҡ„дёҖеҖӢдҫӢеӯҗпјҡ

// Call f64 module
use std::f64;
 
fn main() {
    // Call sqrt function
    let n = f64::sqrt(4.0);
 
    println!("{}", n);
}

еңЁйҖҷеҖӢдҫӢеӯҗдёӯпјҢжҲ‘еҖ‘е‘јеҸ« std еҮҪејҸеә«д№Ӣдёӯзҡ„ f64 жЁЎзө„пјҢд№ӢеҫҢпјҢе°ұеҸҜд»Ҙе‘јеҸ«и©ІжЁЎзө„е…§зҡ„ sqrt еҮҪејҸгҖӮ

ж’°еҜ«жЁЎзө„

еңЁ Rust дёӯпјҢдҪҝз”Ё mod йҖҷеҖӢй—ңйҚөеӯ—дҫҶе»әз«ӢжЁЎзө„гҖӮеҰӮдёӢдҫӢпјҡ

mod english {
    pub fn hello(name: &str) {
        println!("Hello, {}", name);
    }
}
 
mod chinese {
    pub fn hello(name: &str) {
        println!("дҪ еҘҪпјҢ{}", name);
    }
}
 
fn main() {
    // Call modules
    use english;
    use chinese;
 
    // Call two functions with the same name
    english::hello("Michael");
    chinese::hello("йәҘеҸҜ");
}

еңЁжң¬дҫӢдёӯпјҢжҲ‘еҖ‘еңЁе…©еҖӢжЁЎзө„дёӯе®ҡзҫ©дәҶеҗҢеҗҚиҖҢдёҚеҗҢеҠҹиғҪзҡ„еҮҪејҸгҖӮз”ұж–јйҖҷе…©еҖӢеҮҪејҸиў«еҚҖйҡ”дёҚеҗҢзҡ„жЁЎзө„дёӯпјҢдёҚжңғжңүе‘ҪеҗҚиЎқзӘҒзҡ„е•ҸйЎҢгҖӮжЁЎзө„йҷӨдәҶеҚҖйҡ”еҮҪејҸеҗҚзЁұеӨ–пјҢд№ҹжҸҗдҫӣз§ҒжңүеҚҖеЎҠпјҢеңЁжЁЎзө„дёӯзҡ„еҮҪејҸжҲ–зү©д»¶пјҢйңҖд»Ҙ pub й—ңйҚөеӯ—е®Је‘ҠпјҢеҗҰеүҮз„Ўжі•еңЁжЁЎзө„еӨ–дҪҝз”ЁгҖӮ

еҰӮеҗҢжҲ‘еҖ‘е…ҲеүҚзңӢеҲ°зҡ„зҜ„дҫӢпјҢжЁЎзө„д№ҹеҸҜд»Ҙе…§еөҢгҖӮеҰӮдёӢдҫӢпјҡ

mod english {
    pub mod greeting {
        pub fn hello(name: &str) {
            println!("Hello, {}", name);
        }
    }
 
    pub mod farewell {
        pub fn goodbye(name: &str) {
            println!("Goodbye, {}", name);
        }
    }
}
 
fn main() {
    use english;
 
    english::greeting::hello("Michael");
    english::farewell::goodbye("Michael");
}

з”ұжң¬дҫӢеҸҜзҹҘпјҢйҖҸйҒҺжЁЎзө„зҡ„ж©ҹеҲ¶пјҢеҸҜд»ҘеҚ”еҠ©жҲ‘еҖ‘ж•ҙзҗҶеҮҪејҸгҖӮ

е»әз«ӢеҘ—д»¶

еңЁжҲ‘еҖ‘е…ҲеүҚзҡ„зҜ„дҫӢдёӯпјҢжҲ‘еҖ‘е»әз«Ӣзҡ„жҳҜжҮүз”ЁзЁӢејҸе°ҲжЎҲпјҢеҰӮдёӢпјҡ

cargo new --bin myapp

з•¶е·ҘзЁӢиҰҸжЁЎи®ҠеӨ§жҷӮпјҢжҠҠзЁӢејҸзўјеҜ«еҲ°дёҖеҖӢз”ҡиҮіе№ҫеҖӢжӘ”жЎҲдёӯпјҢйғҪжҳҜдёҚеӨӘиҒ°жҳҺзҡ„еҒҡжі•пјҢеҸҜиғҪеӯҳеңЁд»ҘдёӢе•ҸйЎҢпјҡ

Rust жңүиҮӘе·ұзҡ„иҰҸеүҮе’Ңзҙ„е®ҡдҫҶзө„з№”е…¶жЁЎзө„пјӣдҫӢеҰӮпјҡдёҖеҖӢ crate еҢ…жңҖеӨҡеҸҜд»ҘжңүдёҖеҖӢеә« crateпјҢд»»ж„ҸеӨҡеҖӢдәҢйҖІеҲ¶crateгҖҒеҢҜе…ҘиіҮж–ҷеӨҫе…§зҡ„жЁЎзө„зҡ„е…©зЁ®зҙ„е®ҡж–№ејҸзӯүзӯүпјӣ

е…ҲжҠҠдёҖдәӣиЎ“иӘһиӘӘжҳҺдёҖдёӢпјҡ

  • й …зӣ®(Packages)пјҡдёҖеҖӢ Cargo жҸҗдҫӣзҡ„ featureпјҢеҸҜд»Ҙз”ЁдҫҶе»әж§ӢгҖҒжё¬и©Ұе’ҢеҲҶдә«еҢ…пјӣ
  • еҢ…(Crate)пјҡдёҖеҖӢз”ұеӨҡеҖӢжЁЎзө„зө„жҲҗзҡ„жЁ№еҪўзөҗж§ӢпјҢеҸҜд»ҘдҪңзӮәдёүж–№еә«йҖІиЎҢеҲҶзҷјпјҢд№ҹеҸҜд»Ҙз”ҹжҲҗеҸҜеҹ·иЎҢжӘ”жЎҲйҖІиЎҢйҒӢиЎҢпјӣ
  • жЁЎзө„(Module)пјҡеҸҜд»ҘдёҖеҖӢжӘ”жЎҲеӨҡеҖӢжЁЎзө„пјҢд№ҹеҸҜд»ҘдёҖеҖӢжӘ”жЎҲдёҖеҖӢжЁЎзө„пјҢжЁЎзө„еҸҜд»Ҙиў«иӘҚзӮәжҳҜзңҹеҜҰй …зӣ®дёӯзҡ„зЁӢејҸзўјзө„з№”е–®е…ғпјӣ

йҰ–е…ҲпјҢеҢ…(crate) жҳҜ Cargo дёӯзҡ„е®ҡзҫ©пјҢеҹ·иЎҢ cargo new xxxx е°ұжҳҜе»әз«ӢдәҶдёҖеҖӢеҢ…пјҢcrate жҳҜдәҢйҖІеҲ¶(bin)жҲ–еә«(lib)й …зӣ®пјӣ

Rust зҙ„е®ҡпјҡеңЁ Cargo.toml зҡ„еҗҢзҙҡзӣ®йҢ„дёӢпјҡ

  • еҢ…еҗ«src/main.rsжӘ”жЎҲпјҢе°ұжҳҜиҲҮеҢ…еҗҢеҗҚзҡ„дәҢйҖІеҲ¶crateпјӣ
  • еҢ…еҗ«src/lib.rsпјҢе°ұжҳҜиҲҮеҢ…еҗҢеҗҚзҡ„еә«crateпјӣ

дёҖеҖӢеҢ…е…§еҸҜд»ҘжңүеӨҡеҖӢ crateпјҢеӨҡеҖӢcratesе°ұжҳҜдёҖеҖӢжЁЎзө„зҡ„жЁ№еҪўзөҗж§ӢпјӣдҫӢеҰӮпјҢеҰӮжһңдёҖеҖӢеҢ…е…§еҗҢжҷӮеҢ…еҗ«src/main.rsе’Ңsrc/lib.rsпјҢйӮЈйәјд»–е°ұжңүе…©еҖӢcrateпјӣ

еҰӮжһңжғіиҰҒеҢ…еҗ«еӨҡеҖӢдәҢйҖІеҲ¶crateпјҢrustиҰҸе®ҡпјҡйңҖиҰҒе°ҮжӘ”жЎҲж”ҫеңЁsrc/binзӣ®йҢ„дёӢпјҢжҜҸеҖӢжӘ”жЎҲе°ұжҳҜдёҖеҖӢе–®зҚЁзҡ„crateпјҒ

жүҖд»Ҙи·Ҝеҫ‘зҡ„йҒёз”ЁпјҢеҫҖеҫҖйӮ„жҳҜеҸ–жұәж–је ҙжҷҜпјҢд»ҘеҸҠжңӘдҫҶзЁӢејҸзўјзҡ„еҸҜиғҪиө°еҗ‘пјӣ

зөҗж§Ӣй«”е’ҢеҲ—иҲүзҡ„еҸҜиҰӢжҖ§

  • е°Үзөҗж§Ӣй«”иЁӯе®ҡзӮә pubпјҢдҪҶе®ғзҡ„жүҖжңүж¬„дҪҚдҫқз„¶жҳҜз§Ғжңүзҡ„пјӣ
  • е°ҮеҲ—иҲүиЁӯе®ҡзӮә pubпјҢе®ғзҡ„жүҖжңүж¬„дҪҚеүҮе°Үе°ҚеӨ–еҸҜиҰӢпјӣ

  • иҖҢзөҗж§Ӣй«”зҡ„жҮүз”Ёе ҙжҷҜжҜ”ијғиӨҮйӣңпјҢе…¶дёӯзҡ„ж¬„дҪҚд№ҹеҫҖеҫҖйғЁеҲҶеңЁ A иҷ•иў«дҪҝз”ЁпјҢйғЁеҲҶеңЁ B иҷ•иў«дҪҝз”ЁпјҢеӣ жӯӨз„Ўжі•зўәе®ҡжҲҗе“Ўзҡ„еҸҜиҰӢжҖ§пјҢйӮЈзҙўжҖ§е°ұиЁӯе®ҡзӮәе…ЁйғЁдёҚеҸҜиҰӢпјҢе°ҮйҒёж“Үж¬ҠдәӨзөҰзЁӢејҸиЁӯиЁҲеё«пјӣ
