trait з”Ёжі•

е®Је‘Ҡ interface

  • interface иЈЎеҸҜд»Ҙе®ҡзҫ© function жҲ– type
  • interface иЈЎдёҚиғҪеҜҰдҪңеҸҰдёҖеҖӢ interfaceпјҢдҪҶ interface д№Ӣй–“еҸҜд»Ҙз№јжүҝ

  • дҪҝз”Ё trait е®Је‘Ҡ interface
  • еӯӨе…’еҺҹеүҮ (Orphan Rule)
    • иҰҒеҜҰзҸҫжҹҗеҖӢ traitпјҢйҖҷеҖӢ trait еҝ…й ҲиҰҒеңЁз•¶еүҚзҡ„ crate дёӯиў«е®ҡзҫ©
    • з”ЁдҫҶйҒҝе…ҚжЁҷжә–еҮҪејҸеә«пјҢжҲ–еңЁе…¶д»–ең°ж–№иў«е®ҡзҫ©еҘҪзҡ„ trait иў«дҝ®ж”№еҲ°пјҢиҖҢйӣЈд»ҘиҝҪжҹҘ

еҜҰдҪңиҮӘе·ұзҡ„ Add:

trait Add<RHS, Output> {
    fn my_add (self, rhs: RHS) -> Output;
}

impl Add<i32, i32> for i32 {
    fn my_add (self, rhs: i32) -> i32 {
        self + rhs
    }
}

impl Add<u32, i32> for u32 {
    fn my_add (self, rhs: u32) -> i32 {
        (self + rhs) as i32
    }
}

fn main () {
    let (a, b, c, d) = (1i32, 2i32, 3u32, 4u32);
    let x: i32 = a.my_add(b);
    let y: i32 = c.my_add(d);
    println!("{}", x);
    println!("{}", y);
}
